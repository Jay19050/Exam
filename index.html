{
    if(r==n-1)

C++ Stack Program

#include <iostream.h>
#include <conio.h>

class stack
{
private:
    int top, n, *s;

public:
    stack()
    {
        top = -1;
        n = 5;
        s = new int[n];
    }

    void push(int);
    int pop();
    int peep(int);
    void change(int, int);
    void display();
};

void stack::push(int data)
{
    if (top == n - 1)
    {
        cout << "
Stack is Overflow";
        return;
    }
    top = top + 1;
    s[top] = data;
}

int stack::pop()
{
    int data;
    if (top == -1)
    {
        cout << "
Stack is Underflow";
        return 0;
    }
    data = s[top];
    top = top - 1;
    return data;
}

int stack::peep(int p)
{
    if (top - p + 1 < 0)
    {
        cout << "
Stack is Underflow on Peep";
        return 0;
    }
    return s[top - p + 1];
}

void stack::change(int p, int x)
{
    if (top - p + 1 < 0)
    {
        cout << "
Stack is Underflow on Change";
        return;
    }
    s[top - p + 1] = x;
}

void stack::display()
{
    if (top == -1)
    {
        cout << "
Stack is Underflow";
        return;
    }
    for (int i = top; i >= 0; i--)
        cout << endl << s[i];
}

int main()
{
    int ch, ele, p;
    stack obj;

";

";
        cout << "6. EXIT
";
        cout << "Enter Choice: ";
        cin >> ch;

        switch (ch)
        {
        case 1:
            cout << "
Enter Element: ";
            cin >> ele;
            obj.push(ele);
            break;

        case 2:
            ele = obj.pop();
            if (ele != 0)
                cout << "
Deleted Element is: " << ele;
            break;

        case 3:
            cout << "
Enter Position: ";
            cin >> p;
            ele = obj.peep(p);
            if (ele != 0)
                cout << "
Element at Position " << p << " is: " << ele;
            break;

        case 4:
            cout << "
Enter Position: ";
            cin >> p;
            cout << "Enter New Element: ";
            cin >> ele;
            obj.change(p, ele);
            break;

        case 5:
            obj.display();
            break;

        case 6:
            cout << "Exiting...";
            break;

        default:
            cout << "Wrong Choice!";
        }

        getch();
    } while (ch != 6);

    return 0;
}

Python CSV to SQLite3 Program

"""
W.A.P. to convert CSV file to sqlite3 table
"""

import sqlite3
import pandas as pd

# Read CSV file
mydf = pd.read_csv("D:/stud2.csv")
print(mydf)

# Add total marks
mydf["tot"] = mydf["m1"] + mydf["m2"] + mydf["m3"]
print(mydf)

# Add percentage
mydf["pct"] = mydf["tot"] / 3
print(mydf)

# Save to sqlite3
cn = sqlite3.connect("C:/sqlite3/itech2025_10.db")

mydf.to_sql('student', cn, if_exists='replace', index=False)
print("Table Created Successfully")
